
// Example 1 and 2
QQ:=Rationals();
R<x,y,z>:=PolynomialRing(QQ,3);
G0:=(x^2+y^2)*z-2*x*y*(x+y)+(2*x-z)*(2*y-z)*z;
G1:=(x^2+y^2)*z-2*x*y*(x+y);
G:=G0*G1;
NonADE(R,G);

S:=z^4-x^2*y^2; T:=(z^2-x^2)*(z^2-y^2);
f1:=(S+3*T);
f2:=(S+2*T);
f:=f1*f2;
NonADE(R,f);

f:=a*(4*b*(b+a)-a*c)*(a^2*(c-4*b)+b*c*(b-2*a));
NonADE(R,f);
